// O(n^2) time | O(1) space
fun selectionSort(arr: MutableList<Int>) {
    val n = arr.size

    for (i in 0 until n) {
        var min = i

        for (j in (i + 1) until n) {
            if (arr[j] < arr[min]) {
                min = j
            }
        }

        swap(arr, i, min)
    }
}

fun swap(arr: MutableList<Int>, i: Int, j: Int) {
    val temp = arr[i]
    arr[i] = arr[j]
    arr[j] = temp
}
