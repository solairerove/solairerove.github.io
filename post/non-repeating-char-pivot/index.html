// O(n) time | O(n) space
fun nonRepeatingCharPivot(s: String): Int {
    val freq = mutableMapOf<Char, Int>()

    s.forEach { freq[it] = freq.getOrDefault(it, 0) + 1 }

    for (i in s.indices) {
        if (freq[s[i]] == 1) return i
    }

    return -1
}
